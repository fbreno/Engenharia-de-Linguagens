vazio bubble_sort (int vetor[], int n) {
    int k; 
    int j; 
    int aux;

    para (k = 1; k < n; k++) {
        escreva("\n[%d] ", k);

        para (j = 0; j < n - 1; j++) {
            escreva("%d, ", j);

            se (vetor[j] > vetor[j + 1]) {
                aux          = vetor[j];
                vetor[j]     = vetor[j + 1];
                vetor[j + 1] = aux;
            }
        }
    }
}
