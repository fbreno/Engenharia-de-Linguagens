vazio merge(int arr[], int left[], int right[], int leftSize, int rightSize) {
    int i = 0; int j = 0; int k = 0;

    enquanto (i < leftSize && j < rightSize) {
        se (left[i] <= right[j]) {
            arr[k++] = left[i++];
        } senao {
            arr[k++] = right[j++];
        }
    }

    enquanto (i < leftSize) {
        arr[k++] = left[i++];
    }

    enquanto (j < rightSize) {
        arr[k++] = right[j++];
    }
}
